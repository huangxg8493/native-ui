图标库使用说明

1. 引入 SVG Sprite 文件

在页面 HTML 中添加以下代码，加载图标 Sprite：

<script>
  async function loadIconSprites() {
    const files = [
      'assets/icons/ant-design-icons.svg',
      'assets/icons/element-plus-icons.svg',
      'assets/icons/lucide-icons.svg',
      'assets/icons/heroicons.svg',
      'assets/icons/phosphor-icons.svg',
      'assets/icons/tabler-icons.svg',
      'assets/icons/fontawesome-icons.svg'
    ];
    for (const url of files) {
      const resp = await fetch(url);
      const text = await resp.text();
      const div = document.createElement('div');
      div.style.display = 'none';
      div.innerHTML = text;
      document.body.appendChild(div);
    }
  }
  loadIconSprites();
</script>

2. 使用图标

在页面任意位置使用 <use> 引用图标：

<!-- Ant Design 图标 -->
<svg width="16" height="16"><use href="#ant-user"></use></svg>

<!-- Element Plus 图标 -->
<svg width="16" height="16"><use href="#el-user"></use></svg>

<!-- Lucide 图标 -->
<svg width="16" height="16"><use href="#lucide-user"></use></svg>

<!-- Heroicons 图标 -->
<svg width="16" height="16"><use href="#hero-user"></use></svg>

<!-- Phosphor 图标 -->
<svg width="16" height="16"><use href="#ph-user"></use></svg>

<!-- Tabler 图标 -->
<svg width="16" height="16"><use href="#tabler-user"></use></svg>

<!-- Font Awesome 图标 -->
<svg width="16" height="16"><use href="#fa-user"></use></svg>
